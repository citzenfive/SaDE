import subprocess
import ast  # Essencial para a conversão segura de string para lista
import numpy as np # Usado no exemplo para calcular o fitness final

def sua_funcao_evaluate(individual):
    """
    Avalia um indivíduo executando um programa C externo e processando sua saída.

    Esta função espera que o programa C imprima uma lista de números no formato:
    "[valor1, valor2, valor3, valor4]"
    """
    
    # --- Passo 1: Configuração do Comando ---
    # Defina o caminho para o seu programa C já compilado.
    caminho_executavel = "./meu_programa_exec" 
    
    # Constrói o comando como uma lista de strings.
    # É crucial converter cada 'gene' do indivíduo para string.
    try:
        comando = [caminho_executavel] + [str(gene) for gene in individual]
    except Exception as e:
        print(f"ERRO ao construir o comando. O indivíduo é válido? Indivíduo: {individual}. Erro: {e}")
        return (1.0e10,) # Retorna uma penalidade

    # --- Passo 2: Execução e Processamento Seguro ---
    try:
        # Executa o comando e captura a saída. 'check=True' lançará um erro
        # se o programa C retornar um código de saída diferente de 0 (indicando falha).
        resultado_processo = subprocess.run(
            comando, 
            capture_output=True, 
            text=True, 
            check=True
        )
        
        # Pega a string de saída e remove espaços/quebras de linha extras.
        output_str = resultado_processo.stdout.strip()
        
        # --- Passo 3: Parsing da Saída com ast.literal_eval ---
        # Converte a string '[v1, v2, ...]' em uma lista Python de números.
        try:
            valores = ast.literal_eval(output_str)
            # Verifica se o resultado é de fato uma lista, como esperado.
            if not isinstance(valores, list):
                raise TypeError("A saída não foi uma lista.")
        except (ValueError, SyntaxError, TypeError) as e:
            print(f"ERRO: A saída do programa C não pôde ser convertida para uma lista. Erro: {e}")
            print(f"   Saída recebida: '{output_str}'")
            return (1.0e10,)

        # --- Passo 4: Cálculo do Fitness Final a partir dos Valores ---
        # Agora você tem uma lista de números em 'valores'.
        # Você precisa combiná-los em uma única métrica de fitness.
        #
        # **SUBSTITUA A LINHA ABAIXO PELA LÓGICA DO SEU PROBLEMA.**
        # Exemplo: Usar a média dos quatro valores como fitness a ser minimizado.
        fitness_value = np.mean(valores)
        
        # --- Passo 5: Retorno no Formato DEAP ---
        # Retorna o fitness como uma tupla. A vírgula é essencial.
        return (fitness_value,)

    # --- Passo 6: Tratamento de Erros Específicos ---
    except FileNotFoundError:
        print(f"ERRO CRÍTICO: O executável '{caminho_executavel}' não foi encontrado.")
        return (1.0e10,)
    except subprocess.CalledProcessError as e:
        print(f"ERRO CRÍTICO: O programa em C falhou (código de saída {e.returncode}).")
        print(f"   Saída de erro (stderr): {e.stderr.strip()}")
        return (1.0e10,)
    except subprocess.TimeoutExpired:
        print(f"ERRO CRÍTICO: O programa em C demorou demais para responder (timeout).")
        return (1.0e10,)
    except Exception as e:
        # Pega qualquer outro erro inesperado.
        print(f"Ocorreu um erro inesperado durante a avaliação: {e}")
        return (1.0e10,)
